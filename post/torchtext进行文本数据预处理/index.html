torchtext.data.Field(sequential=True, use_vocab=True, init_token=None, eos_token=None, fix_length=None, dtype=torch.int64, preprocessing=None, postprocessing=None, lower=False, tokenize=None, tokenizer_language='en', include_lengths=False, batch_first=False, pad_token='<pad>', unk_token='<unk>', pad_first=False, truncate_first=False, stop_words=None, is_target=False)

这里我挑几个实际应用时常设置的参数进行说明：

  • sequential:是否将数据表示成序列，如果是False不能使用分词；一般对于text使用True，对于label使用False；
  • use_vocab:如果数据已经是数值类型，比如label已经是0,1之类的数据，则设置为False
  • fix_length: 需要设置，长度不足则进行填充，长度超过则进行裁剪
  • tokenize: 如果文本是中文的话，需要定义一个分词函数；函数输入是整个batch的文本，输出是分词后的列表
  • batch_first: 设置为True后，之后从Iterator中输出的数值化后的数据是BatchFirst的；如果不设置，则后来用的时候需要转置一下。
  • include_lengths: 设置为ture之后，每个batch返回的是：（（batchx,batchlen),batchy),如果模型中要用到LSTM的pack_padded形式的forword运算，则在这里需要设置为True，并且，在后面构造Iterator时还要对每个batch进行排序。默认是False，对应batch返回的是（batchx,batchy)
  • stop_words:接受一个停用词列表

class BatchWrapper:
    def __init__(self, dl, x_var, y_var):
        self.dl, self.x_var, self.y_var = dl, x_var, y_var
        # 传入自变量x列表和因变量y列表
    def __iter__(self):
        for batch in self.dl:
            # 在这个封装中只有一个自变量
            x = getattr(batch, self.x_var) 
            if self.y_var is not None:
                y = getattr(batch,self.y_var)
            if self.y_vars is not None: # 把所有因变量cat成一个向量
                temp = [getattr(batch, feat).unsqueeze(1) for feat in self.y_vars]
                y = torch.cat(temp, dim=1).float()
            else:
                y = torch.zeros((1))
                yield (x, y)
            else:
                yield x
    def __len__(self):
        return len(self.dl)
